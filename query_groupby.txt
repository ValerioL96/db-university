1. Contare quanti iscritti ci sono stati ogni anno

SELECT YEAR(enrolment_date) AS anno, COUNT(*) AS numero_di_iscritti
FROM students
GROUP BY anno
ORDER BY anno;


2. Contare gli insegnanti che hanno l'ufficio nello stesso edificio

SELECT office_address, COUNT(*) AS numero_di_insegnanti
FROM teachers
GROUP BY edificio
ORDER BY edificio;


3. Calcolare la media dei voti di ogni appello d'esame

SELECT exam_id, data, AVG(vote) AS media_voti
FROM exams
GROUP BY exam_id, data
ORDER BY data;


4. Contare quanti corsi di laurea ci sono per ogni dipartiment

SELECT departments, COUNT(*) AS numero_di_corsi
FROM courses
GROUP BY departments
ORDER BY departments;
